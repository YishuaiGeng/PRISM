\subsection{Reproducibility Requirements}

\begin{itemize}
    \item \textbf{Python Version}: 3.10+
    \item \textbf{Key Dependencies}:
        \begin{itemize}
            \item z3-solver >= 4.12.0
            \item sentence-transformers (all-MiniLM-L6-v2)
            \item scikit-learn (agglomerative clustering)
            \item OpenAI API >= 0.27 or equivalent
        \end{itemize}
    \item \textbf{Computational Requirements}:
        \begin{itemize}
            \item Offline phase: ~48 GPU-hours (for LLM abstraction and verification)
            \item Online evaluation: ~100 CPU-hours (batch evaluation on test set)
        \end{itemize}
    \item \textbf{Random Seeds}: All experiments use seed=42 for reproducibility.
\end{itemize}
